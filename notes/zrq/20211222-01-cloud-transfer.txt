#[user@desktop]

    branchname=cloud-transfer

    source "${HOME:?}/aglais.env"
    pushd "${AGLAIS_CODE}"

        branchprev=$(git branch --show-current)
        branchnext=$(date '+%Y%m%d')-zrq-${branchname:?}

        git checkout master
        git checkout -b "${branchnext:?}"

        git push --set-upstream 'origin' "$(git branch --show-current)"

    popd
