# Download our kubectl image from blue.
#[root@ansibler]

    imgcloud='iris-gaia-blue'
    imgname='gaia-dmp-ubuntu-2204-kube-v1.26.7'
    imgfile="${imgname}.img"

    imgident=$(
        openstack \
            --os-cloud "${imgcloud}" \
            image show \
                --format json \
                "${imgname}" \
        | jq -r '.id'
        )

    openstack \
        --os-cloud "${imgcloud}" \
        image save \
            "${imgident}" \
    > "${imgfile}"


    ls -al

    >   ....
    >   -rw-r--r--.   1 root root 10G Jan 11 06:13 gaia-dmp-ubuntu-2204-kube-v1.26.7.img
    >   ....
